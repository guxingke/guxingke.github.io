背景

新业务不想使用 nginx 作为 LB, 调研下来, 选择 traefik 作为LB. 目前业务还没有使用容器化相关技术, 所以采用裸机部署.

步骤

基础配置文件

采用 多文件配置方式

[traefikLog]
filePath = "/usr/logs/traefik/traefik.log"
format   = "json"

[accessLog]
filePath = "/usr/logs/traefik/access.log"
format = "json"

[entryPoints]
  [entryPoints.http]
  address = ":80"
  [entrypoints.traefik]
  address = ":8080"
  [entryPoints.https]
  address = ":443"
  [entryPoints.https.tls]
  [[entryPoints.https.tls.certificates]]
  certFile = "/etc/conf/traefik/cert/2020_07_23_api.xxxx.com.pem"
  keyFile  = "/etc/conf/traefik/cert/2020_07_23_api.xxxx.com.key"
  [[entryPoints.https.tls.certificates]]
  certFile = "/etc/conf/traefik/cert/2020_07_23_api-beta.xxxx.com.pem"
  keyFile  = "/etc/conf/traefik/cert/2020_07_23_api-beta.xxxx.com..key"

[api]
  entrypoint = "traefik"
  dashboard = true
  debug = true

[ping]

[file]
  directory= "/etc/conf/traefik/config/"
  watch = true

其中具体 server 的配置放到 config 目录下 示例配置 api.xxxx.com.toml

# forntend

启停脚本

#!/usr/bin/env bash
#
# Startup script for traefik under centos, macos.
set -e

BASE_DIR="dt"

ITEM_NAME=traefik
LOGS="/${BASE_DIR}/logs/usr/$ITEM_NAME"

APP_PID="${LOGS}/${ITEM_NAME}.pid"
APP_HOME="/${BASE_DIR}/dist/sys/${ITEM_NAME}"
APP_EXEC="${APP_HOME}/$ITEM_NAME"
APP_CONF_HOME="/${BASE_DIR}/dist/conf/${ITEM_NAME}"
APP_CONF_FILE="${APP_CONF_HOME}/${ITEM_NAME}.toml"

mkdir -p ${LOGS}

usage()
{
    echo "Usage: ${0##*/} [-d] {start|stop|run|restart|status} "
    exit 1
}

[[ $# -gt 0 ]] || usage

##################################################
# Some utility functions
##################################################
running()
{
  if [[ -f "$1" ]]
  then
    local PID=$(cat "$1" 2>/dev/null) || return 1
    kill -0 "$PID" 2>/dev/null
    return
  fi
  rm -f "$1"
  return 1
}

DEBUG=0
while [[ $1 = -* ]]; do
  case $1 in
    -d) DEBUG=1 ;;
  esac
  shift
done
ACTION=$1
shift


#####################################################
# Must define LOGS
#####################################################
if [[ -z "$LOGS" ]]
then
  echo "Cannot find a LOGS cfg. Please set LOGS." 2>&2
  exit 1
fi


RUN_ARGS=( -c "$APP_CONF_FILE")
RUN_CMD=("$APP_EXEC" ${RUN_ARGS[@]})

#####################################################
# Comment these out after you're happy with what
# the script is doing.
#####################################################
if (( DEBUG ))
then
  echo "APP_HOME     =  $APP_HOME"
  echo "APP_PID      =  $APP_PID"
  echo "RUN_CMD        =  ${RUN_CMD[@]}"
fi

echo ""

##################################################
# Do the action
##################################################
case "$ACTION" in
  start)
    echo -n "Starting ${ITEM_NAME}: "

    if running ${APP_PID}
    then
      echo "Already Running $(cat ${APP_PID})!"
      exit 1
    fi

    "${RUN_CMD[@]}" > "$LOGS/stdout.log" 2>&1 &
    disown $!
    echo $! > "$APP_PID"
    echo "deploy `date`" >> "$LOGS/start.log"
    echo OK

    ;;

  stop)
    echo -n "Stopping ${ITEM_NAME}: "

    if [ ! -f "$APP_PID" ] ; then
      echo "ERROR: no pid found at $APP_PID"
      exit 1
    fi

    PID=$(cat "$APP_PID" 2>/dev/null)
    if [ -z "$PID" ] ; then
      echo "ERROR: no pid id found in $APP_PID"
      exit 1
    fi
    kill "$PID" 2>/dev/null

    TIMEOUT=10
    while running $APP_PID; do
      if (( TIMEOUT-- == 0 )); then
        kill -KILL "$PID" 2>/dev/null
      fi

      sleep 1
    done

    rm -f "$APP_PID"
    echo OK

    ;;

  restart)
    JETTY_SH=$0
    "$JETTY_SH" stop "$@"
    "$JETTY_SH" start "$@"
    ;;


  run)
    echo "Running $ITEM_NAME: "

    if running "$APP_PID"
    then
      echo Already Running $(cat "$APP_PID")!
      exit 1
    fi

    exec "${RUN_CMD[@]}"

    ;;

  status)
    echo "Checking arguments to ${ITEM_NAME}: "
    echo "APP_HOME     =  $APP_HOME"
    echo "APP_PID      =  $APP_PID"
    echo "LOGS         =  $LOGS"
    echo "RUN_CMD      =  ${RUN_CMD[*]}"
    echo

    if running "$APP_PID"
    then
      echo "${ITEM_NAME} running pid=$(< "$APP_PID")"
      exit 0
    fi
    exit 1

    ;;

  *)
    usage

    ;;
esac

exit 0
